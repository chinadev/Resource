swift 配置 (1.4.3-1.4.4成功)

代理节点 192.168.1.101

存储节点 192.168.1.101   #多个存储节点只需要做相应的修改即可，硬盘要自己格式化

---------------------------------------------------------------- 

#!/bin/sh

 

mkdir /etc/swift

mkdir /var/run/swift

 

export PROXY_LOCAL_NET_IP=192.168.2.103

export STORATE_NET_IP=192.168.2.103

export PART_ZONE=5

export BACKUP_NUM=1

export DELETE_TIME=1

 

echo 'deb http://ppa.launchpad.net/openstack-release/2011.3/ubuntu natty main' >> /etc/apt/sources.list

apt-key adv --keyserver keyserver.ubuntu.com --recv-keys 3D1B4472

apt-get install python-software-properties

add-apt-repository ppa:swift-core/release

apt-get update

apt-get install swift openssh-server

apt-get install swift-proxy memcached

chown swift:swift /var/run/swift

chown swift:swift /etc/swift

 

cd /etc/swift

 

#生成swift hash

cat >/etc/swift/swift.conf <<EOF

[swift-hash]

# random unique string that can never change (DO NOT LOSE)

swift_hash_path_suffix = `od -t x8 -N 8 -A n </dev/random`

EOF

 

#swift-proxy配置

cat >/etc/swift/proxy-server.conf <<EOF

[DEFAULT]

cert_file = /etc/swift/cert.crt

key_file = /etc/swift/cert.key

bind_port = 8080

workers = 8

user = swift

 

[pipeline:main]

pipeline = healthcheck cache tempauth proxy-server

 

[app:proxy-server]

use = egg:swift#proxy

allow_account_management = true

account_autocreate = true

 

[filter:tempauth]

use = egg:swift#tempauth

user_system_root = testpass .admin https://$PROXY_LOCAL_NET_IP:8080/v1/AUTH_system

 

[filter:healthcheck]

use = egg:swift#healthcheck

 

[filter:cache]

use = egg:swift#memcache

 

memcache_servers = $PROXY_LOCAL_NET_IP:11211

EOF

 

perl -pi -e "s/-l 127.0.0.1/-l $PROXY_LOCAL_NET_IP/" /etc/memcached.conf

service memcached restart

 

openssl req -new -x509 -nodes -out cert.crt -keyout cert.key

 

rm -f *.builder *.ring.gz backups/*.builder backups/*.ring.gz

 

#创建账户环

swift-ring-builder account.builder create $PART_ZONE $BACKUP_NUM $DELETE_TIME  

#添加存储账户到环，这里仅仅一个存储节点，如果多个，仍按这个格式添加

swift-ring-builder account.builder add z1-$STORATE_NET_IP:6012/sdb1 100

#确认账户环内容

swift-ring-builder account.builder

 

#重平衡环

swift-ring-builder account.builder rebalance

#创建容器环

swift-ring-builder container.builder create $PART_ZONE $BACKUP_NUM $DELETE_TIME

#添加存储容器到环，这里仅仅是一个，如果多个，请仍按这个格式添加

swift-ring-builder container.builder add z1-$STORATE_NET_IP:6011/sdb1 100

#确认容器环

swift-ring-builder container.builder

 

#重平衡容器环

swift-ring-builder container.builder rebalance

#创建对象环

swift-ring-builder object.builder create $PART_ZONE $BACKUP_NUM $DELETE_TIME

#添加存储对象到环

swift-ring-builder object.builder add z1-$STORATE_NET_IP:6010/sdb1 100

#确认对象环

swift-ring-builder object.builder

#重平衡对象环

swift-ring-builder object.builder rebalance

 

#启动代理服务

swift-init proxy restart

#!/bin.sh

 

mkdir /etc/swift

mkdir /var/run/swift

 

export PROXY_LOCAL_NET_IP=192.168.2.103

export STORAGE_LOCAL_NET_IP=192.168.2.103

 

 

echo 'deb http://ppa.launchpad.net/openstack-release/2011.3/ubuntu natty main' >> /etc/apt/sources.list

apt-key adv --keyserver keyserver.ubuntu.com --recv-keys 3D1B4472

apt-get install python-software-properties openssh-server

add-apt-repository ppa:swift-core/release

apt-get update

chown swift:swift /var/run/swift

chown swift:swift /etc/swift

apt-get install swift-account swift-container swift-object xfsprogs

 

cd /etc/swift

scp $PROXY_LOCAL_NET_IP:/etc/swift/swift.conf /etc/swift

 

#mkfs.xfs  -i size=1024 /dev/sdb1

echo "/dev/sdb1 /srv/node/sdb1 xfs noatime,nodiratime,nobarrier,logbufs=8 0 0" >> /etc/fstab

mkdir -p /srv/node/sdb1

mount /srv/node/sdb1

chown -R swift:swift /srv/node

 

cat >/etc/rsyncd.conf <<EOF

uid = swift

gid = swift

log file = /var/log/rsyncd.log

pid file = /var/run/rsyncd.pid

address = $STORAGE_LOCAL_NET_IP

 

[account6012]

max connections = 2

path = /srv/node/sdb1

read only = false

lock file = /var/lock/account6012.lock

 

[container6011]

max connections = 2

path = /srv/node/sdb1

read only = false

lock file = /var/lock/container6012.lock

 

[object6010]

max connections = 2

path = /srv/node/sdb1

read only = false

lock file = /var/lock/object6010.lock

EOF

 

perl -pi -e 's/RSYNC_ENABLE=false/RSYNC_ENABLE=true/' /etc/default/rsync

service rsync start

 

mkdir -p /etc/swift/account-server

mkdir -p /etc/swift/container-server

mkdir -p /etc/swift/object-server

 

cat > /etc/swift/account-server/1.conf <<EOF

[DEFAULT]

devices = /srv/node/sdb1

mount_check = false

bind_ip=$STORAGE_LOCAL_NET_IP

bind_port = 6012

user = swift

worker=2

 

[pipeline:main]

pipeline = account-server

 

[app:account-server]

use = egg:swift#account

 

[account-replicator]

 

[account-auditor]

 

[account-reaper]

EOF

 

cat >/etc/swift/container-server/1.conf <<EOF

[DEFAULT]

devices = /srv/node/sdb1

mount_check = false

bind_port = $STORAGE_LOCAL_NET_IP

bind_port = 6011

user = swift

 

[pipeline:main]

pipeline = container-server

 

[app:container-server]

use = egg:swift#container

 

[container-replicator]

 

[container-updater]

 

[container-auditor]

 

[container-sync]

EOF

 

cat > /etc/swift/object-server/1.conf <<EOF

[DEFAULT]

devices = /srv/node/sdb1

mount_check = false

bind_ip = $STORAGE_LOCAL_NET_IP

bind_port = 6010

user = swift

 

[pipeline:main]

pipeline = object-server

 

[app:object-server]

use = egg:swift#object

 

[object-replicator]

vm_test_mode = yes

 

[object-updater]

 

[object-auditor]

EOF

 

#更改次文件夹属主为swift（这些程序都以swift用户运行）

chown -R swift:swift /etc/swift

 

#启动所有服务 

swift-init all restart
